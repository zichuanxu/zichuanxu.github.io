## 题目描述

将 `n` 个皇后放置在 `n × n` 棋盘上，使任意两个皇后不能互相攻击（同行、同列、同对角线均不可），返回所有方案。

**示例：** `n = 4` → 2 种摆法。

## 思路分析

**核心思想：** 逐行放置皇后，用三个集合记录被占用的列、两条对角线（`row+col` 与 `row-col`），冲突则跳过。

**详细步骤：**

1. 按行递归；`row == n` 时记录当前棋盘。
2. 每列尝试：若列或两对角线已被占则跳过。
3. 放置并标记，递归下一行，回溯时撤销标记。

```go
func solveNQueens(n int) [][]string {
    res := [][]string{}
    cols := make([]bool, n)
    diag1 := make([]bool, 2*n) // row + col
    diag2 := make([]bool, 2*n) // row - col + n
    board := make([][]byte, n)
    for i := range board {
        board[i] = make([]byte, n)
        for j := range board[i] {
            board[i][j] = '.'
        }
    }
    var backtrack func(row int)
    backtrack = func(row int) {
        if row == n {
            snapshot := make([]string, n)
            for i := range board {
                snapshot[i] = string(board[i])
            }
            res = append(res, snapshot)
            return
        }
        for col := 0; col < n; col++ {
            if cols[col] || diag1[row+col] || diag2[row-col+n] {
                continue
            }
            cols[col], diag1[row+col], diag2[row-col+n] = true, true, true
            board[row][col] = 'Q'
            backtrack(row + 1)
            board[row][col] = '.'
            cols[col], diag1[row+col], diag2[row-col+n] = false, false, false
        }
    }
    backtrack(0)
    return res
}
```

```python
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        res = []
        cols, diag1, diag2 = set(), set(), set()
        board = [['.'] * n for _ in range(n)]
        def backtrack(row):
            if row == n:
                res.append([''.join(r) for r in board])
                return
            for col in range(n):
                if col in cols or (row + col) in diag1 or (row - col) in diag2:
                    continue
                cols.add(col); diag1.add(row + col); diag2.add(row - col)
                board[row][col] = 'Q'
                backtrack(row + 1)
                board[row][col] = '.'
                cols.discard(col); diag1.discard(row + col); diag2.discard(row - col)
        backtrack(0)
        return res
```
